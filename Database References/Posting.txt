    ID              SERIAL PRIMARY KEY,
    Name            varchar(255),
    Category        category,
    Status          status,

CREATE TYPE privilege AS ENUM ('guest', 'user', 'admin');

CREATE TABLE Users (
    ID              SERIAL PRIMARY KEY,
    Username        varchar(255),
    Email           varchar(255),
    Password	    varchar(255),
    Privilege	    privilege,
    Postings        int[]
);
