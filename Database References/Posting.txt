DROP TABLE Posting;
DROP TABLE Users;
DROP TYPE category;
DROP TYPE status;
DROP TYPE privilege;

CREATE TYPE category AS ENUM ('Nature', 'Animals');
CREATE TYPE status AS ENUM ('Pending', 'Approved', 'Rejected');

CREATE TABLE Posting (
    ID              SERIAL PRIMARY KEY,
    Name            varchar(255),
    Category        category,
    Status          status,
    UserID          int,
    Image           varchar(255)
);

SELECT * FROM Posting;

CREATE TYPE privilege AS ENUM ('guest', 'user', 'admin');

CREATE TABLE Users (
    ID              SERIAL PRIMARY KEY,
    Username        varchar(255),
    Email           varchar(255),
    Password	    varchar(255),
    Privilege	    privilege,
    Postings        int[]
);
